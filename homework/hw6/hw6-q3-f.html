L = 1.0
D = 1.0
pi = numpy.pi

dx = 0.001 # step length
nx = int(L/dx) + 1 # steps number
x0 = numpy.linspace(0.0, L, num=nx)

T = [0.001, 0.01, 0.1, 10.0] # different time value
In [3]:
x = x0.copy()
f_history = [x0.copy()]
for t in T:
    f1 = x / L - (x / L)**2
    temp = 0.0
    for n in range(1, 2001):
        temp = temp + ((-1)**n - 1) / (n**3) * numpy.sin(n*pi*x/L) * numpy.exp(-D*((n*pi/L)**2)*t)
    f2 = 4 / (pi**3) * temp
    f = f1 + f2
    f_history.append(f)
